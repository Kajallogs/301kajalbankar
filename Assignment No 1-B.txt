list = [5,9,10,20,16,25,40]
print(list)


list.insert(1,3)        #insert 3 at index 1
print(list)


list.remove(40)            #remove object 
print(list)


list.append(50)           #add object at the end of list
print(list)


list.sort()              # sort the list
print(list)


list.pop(2)              #delete the element 
print(list)   


print(list[::-1])      #does not modify the original list


list.reverse()         #reverse list
